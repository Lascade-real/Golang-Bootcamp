func checkType(x interface{}) {
	switch x.(type) {
	case int:
		fmt.Println("It's an integer")
	case int32:
		fmt.Println("It's an integer 32")
	case float64:
		fmt.Println("It's float")
	case string:
		fmt.Println("String value")
	default:
		fmt.Println("unknown type")
	}
}
