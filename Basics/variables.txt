package basics

import "fmt"

//Global scope variables
var middleName = "Kane"
var firstName = "Connor"

//function with variable also named firstName
func printName(firstName string) {
	fmt.Println(firstName)

}

func main() {

	//variables declared within main
	var age int  //default value is always 0 when not defined
	var name string = "Makoto"
	var name_2 string = "Jane"
	lastName := "Smith"  //short form declaration 

	fmt.Println(age)
	fmt.Println(name)
	fmt.Println(name_2)
	fmt.Println(lastName)
	fmt.Println(middleName)
	fmt.Println(firstName)

	printName("Marco") //correctly prints printName function

}
